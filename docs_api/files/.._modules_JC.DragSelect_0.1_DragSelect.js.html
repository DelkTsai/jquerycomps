    function intersectRect( r1, r2 ) {
        return !(
                    r2.x > ( r1.x + r1.width ) || 
                    ( r2.x + r2.width ) < r1.x || 
                    r2.y > ( r1.y + r1.height ) ||
                    ( r2.y + r2.height ) < r1.y
                );
    }

    function pointToRect( _p1, _p2 ){

        if( _p1 && _p2 ){

                _r.x = _p1.x;
                _r.width = _p2.x - _p1.x;

                _r.x = _p2.x;
                _r.width = _p1.x - _p2.x;

            if( _p1.y < _p2.y ){
                _r.y = _p1.y;
                _r.height = _p2.y - _p1.y;
            }else{
                _r.y = _p2.y;
                _r.height = _p1.y - _p2.y;

            _r.top = _r.y;
